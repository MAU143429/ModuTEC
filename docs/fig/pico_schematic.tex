\definecolor{c3v3}{RGB}{0,90,200}
\definecolor{cgnd}{RGB}{20,135,30}
\definecolor{csig}{RGB}{160,0,0}

\begin{circuitikz}[
    american,
    scale      = 1.0,
    transform shape,
    line width = 0.65pt,
    font       = \small,
]

% ════════════════════════════════════════════════════════════════
%  LAYOUT  (compacted vertical distances)
%
%  Pico box:  (0,0)–(16,3)
%  Top pin stubs: y=3 → y=3.35
%
%  Pot positions (same x, shorter vertical span):
%    bottom (GND) y=4.5   top (3V3) y=7.1
%    Pot1 x=4.5   Wiper → x=5.3
%    Pot2 x=7.5   Wiper → x=8.3
%    Pot3 x=10.5  Wiper → x=11.3
%
%  3V3 rail: y=7.8  (was 9.8)
%  GND rail: y=-1.3 (was -2.0)
%  Switch S1: y=4.5..6.0 (was 4.8..6.5)
% ════════════════════════════════════════════════════════════════

% ── PICO BOX ────────────────────────────────────────────────────
\draw[thick, rounded corners=4pt]
    (0,0) rectangle (16,3);
\node[align=center] at (8,1.5)
    {\normalsize\bfseries Raspberry Pi Pico};

% Top pin stubs  (y=3 → y=3.35)
\foreach \x/\lbl in {
    1.8/{3V3},
    5.3/{GP28\_A2},
    8.3/{GP27\_A1},
    11.3/{GP26\_A0},
    14.0/{GP16}
}{
    \draw (\x,3) -- (\x,3.35);
    \node[above, font=\ttfamily\scriptsize, rotate=60, anchor=west]
        at (\x,3.3) {\lbl};
}

% Bottom GND stub
\draw (8,0) -- (8,-0.35);
\node[below, font=\ttfamily\scriptsize] at (8,-0.35) {GND};

% Cosmetic side pins
\foreach \y in {0.4,0.9,1.4,1.9,2.4,2.9}{
    \draw (0,\y)  -- (-0.22,\y);
    \draw (16,\y) -- (16.22,\y);
}

% ── 3V3 RAIL  (blue) ────────────────────────────────────────────
\draw[c3v3, thick]
    (1.8,3.35) -- (1.8,7.8) -- (10.5,7.8);

% Taps to each pot top pin (y=7.1)
\foreach \x in {4.5,7.5,10.5}{
    \draw[c3v3, thick] (\x,7.8) -- (\x,7.1);
    \fill[c3v3] (\x,7.8) circle (2.2pt);
}

% ── GND RAIL  (green) ───────────────────────────────────────────
\draw[cgnd, thick]
    (8,-0.35) -- (8,-1.3)
    (4.5,-1.3) -- (14.5,-1.3);
\node[ground, cgnd] at (8,-1.3) {};

% Taps from GND rail up to pot bottom pins (y=4.5)
\foreach \x in {4.5,7.5,10.5}{
    \draw[cgnd, thick] (\x,-1.3) -- (\x,4.5);
    \fill[cgnd] (\x,-1.3) circle (2.2pt);
}

% GND tap for switch at x=14.5
\draw[cgnd, thick] (14.5,-1.3) -- (14.5,5.8);
\fill[cgnd] (14.5,-1.3) circle (2.2pt);

% ── POTENTIOMETERS  (vertical, mirror → wiper exits right) ──────
%  bottom = GND (y=4.5)   top = 3V3 (y=7.1)

% ---- Pot 1  (ADC2 / GP28) ----
\draw (4.5,4.5) to[pR, n=p1, mirror] (4.5,7.1);
\node[left=4pt, font=\footnotesize\bfseries, align=right] at (4.1,6.5)
    {Pot 1 \\ 10\,k$\Omega$};

% Wiper → right to x=5.3 → straight down to Pico
\draw[csig, thick]
    (p1.wiper) -- (5.3,5.6) -- (5.3,3.35);
\fill[csig] (5.3,3.35) circle (2pt);

% ---- Pot 2  (ADC1 / GP27) ----
\draw (7.5,4.5) to[pR, n=p2, mirror] (7.5,7.1);
\node[left=4pt, font=\footnotesize\bfseries, align=right] at (7.1,6.5)
    {Pot 2 \\ 10\,k$\Omega$};

\draw[csig, thick]
    (p2.wiper) -- (8.3,5.6) -- (8.3,3.35);
\fill[csig] (8.3,3.35) circle (2pt);

% ---- Pot 3  (ADC0 / GP26) ----
\draw (10.5,4.5) to[pR, n=p3, mirror] (10.5,7.1);
\node[left=4pt, font=\footnotesize\bfseries, align=right] at (10.1,6.5)
    {Pot 3 \\ 10\,k$\Omega$};

\draw[csig, thick]
    (p3.wiper) -- (11.3,5.6) -- (11.3,3.35);
\fill[csig] (11.3,3.35) circle (2pt);

% ── SWITCH S1  (SPST-NO) ────────────────────────────────────────
\draw[thick] (14.0,3.35) -- (14.0,4.2);
\draw (14.0,4.2) to[spst] (14.0,5.8);

% Top terminal → GND rail (green) via x=14.5
\draw[cgnd, thick] (14.0,5.8) -- (14.5,5.8);

\node[right=4pt, font=\footnotesize\bfseries] at (14.0,5.1) {S1};
\node[right=4pt, font=\footnotesize\itshape]  at (14.0,4.5) {AM / ASK};

\end{circuitikz}
